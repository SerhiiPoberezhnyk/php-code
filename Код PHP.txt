<?php
// Get the current date
$currentDate = date('Y-m-d');
// Determining the season
function getSeason($date) {
    $month = intval(date('m', strtotime($date)));
    switch ($month) {
        case 12:
        case 1:
        case 2:
            return 'winter';
        case 3:
        case 4:
        case 5:
            return 'spring';
        case 6:
        case 7:
        case 8:
            return 'summer';
        default:
            return 'autumn';
    }
}
// Get the season for the current date
$currentSeason = getSeason($currentDate);
// Display the name of the season
echo "<h1>Season: $currentSeason</h1>";
// Change the background depending on the season
$background = '';
switch ($currentSeason) {
    case 'winter':
        $background = 'winter.jpg'; // path to winter background
        break;
    case 'spring':
        $background = 'spring.jpg'; // path to spring background
        break;
    case 'summer':
        $background = 'summer.jpg'; // path to summer background
        break;
    case 'autumn':
        $background = 'autumn.jpg'; // path to autumn background
        break;
}
// Deriving styles for changing the background
echo "<style>body {background-image: url('$background');}</style>";
?>
